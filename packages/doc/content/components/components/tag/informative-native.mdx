### Usage

```javascript type=expo
  <Tag.Informative mv="medium">default</Tag.Informative>
  <Tag.Informative variant="success">
    default with custom icon
  </Tag.Informative>

  <Tag.Informative mv="medium" variant="success">success</Tag.Informative>
  <Tag.Informative variant="success" icon={Building}>
    success with custom icon
  </Tag.Informative>

  <Tag.Informative marginVertical="medium" variant="informative" >informative</Tag.Informative>
  <Tag.Informative icon={Building}>informative with custom icon</Tag.Informative>

  <Tag.Informative mv="medium" variant="informative" small>informative small</Tag.Informative>
  <Tag.Informative icon={Building} small>informative small with custom icon</Tag.Informative>

  <Tag.Informative variant="attention">attention</Tag.Informative>
  <Tag.Informative variant="attention" icon={Building}>
    attention with custom icon
  </Tag.Informative>
```
